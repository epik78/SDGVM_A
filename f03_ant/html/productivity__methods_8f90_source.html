865 ! assj2 :: productivity_methods !
866 ! ----------------------------- !
867 ! !
868 ! ASSJ computes the assimilation rate by forming a cubic equation !
869 ! from the equations defining assimilation rate, stomatal conductance, !
870 ! CO2 partial pressure (assimilation diffusion equaiton) and the !
871 ! carboxylation rate. The cubic equation is solved and the largest !
872 ! real root is assumed to be the assimilation rate. The variables here !
873 ! correspond to the Doly paper, in the main program they are such that !
874 ! po,pa,j,g0,g1,rh,kg,tau,rd !
875 ! = oi,ca,j(i),c1,c2,rh,kg,tau,rd !
876 ! !
877 ! SUBROUTINE assj2(a,w,pc,gs,po,pa,j,g0,g1,rh,kg,tau,rd,acheck) !
878 ! !
879 !----------------------------------------------------------------------!
880 subroutine assj2(a,w,pc,gs,po,pa,j,g0,g1,rh,kg,tau,rd,acheck)
881 !**********************************************************************!
882 real(dp) :: a,w,pc,gs,po,pa,j,g0,g1,rh,kg,tau,rd,acheck,r2q3,arg1, &
883  ax,bx,cx,dx,ex,fx,gx,hx,p,q,r,qt,rt,th,a1,a2,a3,pi,at,bt,sc
884 !----------------------------------------------------------------------!
885 
886 pi = atan(1.0)*4.0
887 sc = 1000000.0
888 
889 !----------------------------------------------------------------------!
890 ! 'a2,b2,c2,d2' are such that 'w=(a2a + b2)/(c2a + d2)'. !
891 !----------------------------------------------------------------------!
892 ax = j*tau*pa*(kg*g1*rh - 160.0)
893 bx = j*tau*pa*kg*g0*pa
894 cx = 4.0*pa*kg*tau*g1*rh - 640.0*pa*tau + 4.0*po*kg*g1*rh
895 dx = 4.0*pa**2.0*kg*tau*g0 + 4.0*po*kg*g0*pa
896 
897 !----------------------------------------------------------------------!
898 ! 'ex,fx,gx,hx' are such that '0.5po/(taupc)-1=(exa+fx)/(gxa+hx)'. !
899 !----------------------------------------------------------------------!
900 ex =-tau*pa*kg*g1*rh + 160.0*tau*pa + 0.5*po*kg*g1*rh
901 fx =-tau*pa**2.0*kg*g0 + 0.5*po*kg*g0*pa
902 gx = tau*pa*(kg*g1*rh - 160.0)
903 hx = tau*pa**2.0*kg*g0
904 
905 !----------------------------------------------------------------------!
906 ! Cubic coefficients 'p,q,r'. !
907 !----------------------------------------------------------------------!
908 p = sc*rd + (sc*ax*ex + cx*hx + dx*gx)/(cx*gx)
909 q = (sc*rd*cx*hx + sc*rd*dx*gx + sc*ax*fx + sc*bx*ex + dx*hx)/(cx*gx)
910 r = (sc*rd*dx*hx + sc*bx*fx)/(cx*gx)
911 
912 !----------------------------------------------------------------------!
913 ! Sove the cubic equaiton to find 'a'. !
914 !----------------------------------------------------------------------!
915 qt = (p**2 - 3.0*q)/9.0
916 rt = (2.0*p**3 - 9.0*p*q + 27.0*r)/54.0
917 
918 r2q3 = rt**2 - qt**3
919 
920 if (r2q3<0.0) then
921  arg1 = rt/qt**1.5
922  if (arg1>1.0) arg1 = 1.0
923  if (arg1<-1.0) arg1 = -1.0
924  th = acos(arg1)
925  a1 =-2.0*qt**0.5*cos(th/3.0) - p/3.0
926  a2 =-2.0*qt**0.5*cos((th + 2.0*pi)/3.0) - p/3.0
927  a3 =-2.0*qt**0.5*cos((th + 4.0*pi)/3.0) - p/3.0
928  a = a1
929  if (a2>a) a = a2
930  if (a3>a) a = a3
931 else
932  at =-rt/abs(rt)*(abs(rt) + r2q3**0.5)**(1.0/3.0)
933  if (abs(at)>1e-6) then
934  bt = qt/at
935  else
936  bt = 0.0
937  endif
938  a1 = at + bt - p/3.0
939 ! a2 =-0.5*(at + bt) - p/3.0 + i*3.0**0.5*(at - bt)/2.0
940 ! a3 =-0.5*(at + bt) - p/3.0 - i*3.0**0.5*(at - bt)/2.0
941  a = a1
942 endif
943 
944 !----------------------------------------------------------------------!
945 ! Compute 'gs,pc,w' corresponding to 'a'. !
946 !----------------------------------------------------------------------!
947 gs = (g0 + g1*a*rh/pa)*kg
948 pc = pa - a*160.0/gs
949 w = j*pc/4.0/(pc + po/tau)
950 acheck = (w*(1.0 - 0.5*po/(tau*pc)) - rd)*sc
951 
952 end subroutine assj2
