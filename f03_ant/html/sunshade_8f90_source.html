60 subroutine goudriaanslaw_ant(lyr,lai,beamrad,diffrad,fsunlit,qsunlit, &
61  fshade,qshade,can_clump,cos_zen,s070607)

66 logical :: s070607
67 !----------------------------------------------------------------------!
68 
69 if (.not.(s070607)) then
70 
71 ! Goudriaan's Law as described in Walker ... from Spitters 1986 and
72 ! Wang 2003 Func. Plant Biol.
73 ! all below values are for visible wavelengths
74 
75 ! extinction coefficents of direct & diffuse radiation
76 ! these account for both canopy clumping and solar zenith angle, also
77 ! inform by Bodin & Franklin 2012 GMD
78 ! transmittance equals reflectance
79  leafscattering = 2.0*0.075
80  m = sqrt(1.0 - leafscattering)
81  gbeam = 0.5*can_clump
82  kbeam = gbeam/cos_zen
83  kbeamprime = m*kbeam
84 ! kdiff is Gbeam * can_clump / cos_zen integrated over zenith angle 0 to pi/2
85  kdiff = 0.8*can_clump
86  kdiffprime = m*kdiff
87 
88 ! calculate albedos
89  rhosoil = 0.15
90  rhoh = (1.0 - m)/(1.0 + m)
91  rhobeam_can = rhoh*2.0*kbeam/(kbeam+kdiff)
92  rhodiff_can = 4.0*gbeam*rhoh*(gbeam*(log(gbeam)-log(gbeam+kdiff))/kdiff**2 + 1.0/kdiff)
93  rhobeam = rhobeam_can + (rhosoil-rhobeam_can)*exp(-2.0*kbeamprime*lai)
94  rhodiff = rhodiff_can + (rhosoil-rhodiff_can)*exp(-2.0*kdiffprime*lai)
95 
96 ! calculate absorbed direct & diffuse radiation
97  qshade = (1.0-rhodiff)*diffrad*kdiffprime*exp(-kdiffprime*lyr) &
98  + (1.0-rhobeam)*beamrad*kbeamprime*exp(-kbeamprime*lyr) &
99  - (1.0-leafscattering)*beamrad*kbeam*exp(-kbeam*lyr)
100 
101  if (qshade<0.0) qshade = 0
102  qsunlit = (1.0-leafscattering)*kbeam*beamrad + qshade
103 
104 ! calculate fraction sunlit vs shaded leaves
105  fsunlit = exp(-kbeam*lyr)
106  fshade = 1.0 - fsunlit
107 
108 else
109 
110 ! value from Wang
111  rhosoil = 0.15
112 ! leaf transmittance and reflectance are equal. Then albedo is the double
113  leafscattering = 2.0*0.075
114 
115 ! extinction coefficent of the direct beam
116  kbeam = 0.5*can_clump/cos_zen
117 ! extinction coefficient of diffuse radiations
118  kdiffprime = 0.66
119 ! I found somewhere an approximate relation: kbeam=0.75*kdiff
120 
121  m = sqrt(1.0 - leafscattering)
122  kbeamprime = m*kbeam
123 
124  rhobeam_can = 2.0*kbeam/(kbeam + kdiffprime)*(1.0 - m)/(1.0 + m)
125  rhobeam = rhobeam_can + (rhosoil - rhobeam_can)*exp(-kbeam*lai)
126 ! ... hum, I don't know if albedodiff is really different or not... so equal
127  rhodiff = rhobeam
128 
129  qshade = (1.0 - rhodiff)*diffrad*kdiffprime*exp(-kdiffprime*lyr) &
130  +(1.0 - rhobeam)*beamrad*kbeam*exp(-kbeam*lyr) &
131  -(1.0 - leafscattering)*beamrad*kbeamprime*exp(-kbeamprime*lyr)
132  if (qshade<0.0) qshade = 0.0
133
