55 ! !
56 !----------------------------------------------------------------------!
57 !> @brief Calculates total,direct and diffuse radiation for the day
58 !! @details Units oof outputs are mol/m2/sec
59 !! Needs interpolated daily values of cloud cover instead
60 !! of the same one for all the days in the month
61 !! @author Mark Lomas
62 !! @date Feb 2006
63 !----------------------------------------------------------------------!
64 subroutine pfd(lat,day,hrs,cloud,direct,diffuse,total)
65 !**********************************************************************!
66 implicit none
67 real(dp) :: hrs !< day length in hours
68 real(dp) :: del !< solar declination
69 real(dp) :: toa !< solar flux density average over day
70 real(dp) :: rlat !< latitude in radians
71 real(dp) :: dawn_angle !< Hour angle in radians it will integrate for
72 real(dp) :: total !< Average direct and diffuse radiation in mole/m2/sec
73 real(dp) :: direct !< Average direct radiation in mole/m2/sec
74 real(dp) :: diffuse !< Average diffuse radiation in mole/m2/sec
75 real(dp) :: lat,cloud,diffprop,alpha,beta, &
76  sigma,coscst,clearness
77 integer :: day
78 real(dp), parameter :: conv = 1.74532925e-2, pi = 3.1415927
79 !----------------------------------------------------------------------!
80 
81 if (hrs>1e-6) then
82  del = -23.4*cos(conv*360.0*(day+10.0)/365.0)
83  del = del*conv
84  rlat = lat*conv
85 
86 ! solar radiation in watts/m2 average over the day
87 ! (between sun rise en sun set)
88 
89  dawn_angle = (hrs/2.0)*(2.0*pi/24.0)
90 
91 !Top of atmosphere irradiance, average over the day (daylength).
92 ! 1370 = solar constant in W/m2
93  toa = 1370.0*24.0/(pi*hrs)*(sin(rlat)*sin(del)*dawn_angle + &
94  cos(rlat)*cos(del)*sin(dawn_angle))
95 
96 !Account for cloud cover
97 
98  coscst = 1.0-1.3614*cos(rlat)
99  alpha = 18.0 - 64.884*coscst
100  alpha = alpha*4.1842*1.0d4 ! convert from cal/cm to J/m2
101  alpha = alpha/(24.0*3600.0) ! convert into w/m2
102 
103  beta = 0.682 - 0.3183*coscst
104 
105  sigma = 0.02*log(max(cloud,0.001))+0.03259
106 
107 !Calculate the total irradiance
108  total = toa*(beta-sigma*cloud*10.0)-alpha
109  if (total<0.0) total=0.0
110 
111  clearness= total/toa
112 
113 !Calculate diffuse irradiance (from Forest ETP WG)
114 
115  if(clearness<0.07) then
116  diffprop = 1.0
117  else if(clearness<0.35) then
118  diffprop = 1.0-2.3*(clearness-0.07)**2
119  else if(clearness<0.75) then
120  diffprop = 1.33-1.46*clearness
121  else
122  diffprop = 0.23
123  endif
124 
125  diffuse=total*diffprop
126 
127 ! PAR is about 48% of the total irradiance. A better formula could be
128 ! found.
129  total = 0.48*total
130  diffuse = 0.48*diffuse ! not sure this is really right (Ghislain)
131 
132 !Convert in mol/m2/sec 1W/m2=4.6e-6 mol/s/m2 for PAR radiation
133 
134  total = 4.6d-6*total
135  diffuse = 4.6d-6*diffuse
136  direct= total - diffuse
137 else
138  total=0.0
139  diffuse=0.0
140  direct=0.0
141 endif
142 
143 end subroutine pfd

164 subroutine pfd_ant(lat,day,hrs,cloud,direct,diffuse,total,swr, &
165  read_par,subd_par,t,total_t,calc_zen,cos_zen)
166 !**********************************************************************!
167 REAL(dp) :: lat,hrs,del,rlat,toa,cloud,diffprop,alpha,beta
168 REAL(dp) :: sigma,dawn_angle,coscst,direct,diffuse,total,clearness
169 REAL(dp) :: swr,ts,h,toa_h,tem,dJ_R,dJ_K,cos_zen
170 INTEGER :: day,t,total_t
171 real(dp) :: conv = 1.74532925e-2
172 real(dp) :: pi = 3.1415927
173 real(dp) :: solar_const = 1370.0 ! solar constant (W/m2)
174 logical :: read_par,subd_par,calc_zen
175 !----------------------------------------------------------------------!
176 
177 if (hrs>1e-6) then
178 ! declination angle
179  del = -23.4*cos(conv*360.0*(day+10.0)/365.0)
180  del = del*conv
181  rlat = lat*conv
182 
183 ! calculate dawn angle
184  dawn_angle = (hrs/2.0)*(2.0*pi/24.0)
185 
186 ! calculate daytime mean cos of the zenith angle
187  cos_zen = 1/dawn_angle*(sin(rlat)*sin(del)*dawn_angle + &
188  cos(rlat)*cos(del)*sin(dawn_angle) )
189 
190 ! top of atmosphere irradiance
191 ! - watts/m2 average over the day
192 ! - between sun rise and sun set
193 ! toa = solar_const * cos_zen ! daytime mean irradiance (W/m2)
194  toa = solar_const *24.0/(pi*hrs)*(sin(rlat)*sin(del)*dawn_angle + &
195  cos(rlat)*cos(del)*sin(dawn_angle))
196 
197 ! calculate the mean daylight top of canopy (total) shortwave radiation
198  if (read_par) then
199 !read from the input dataset as mean SWR over 24hrs
200 ! - scale to mean over daylight hours only
201  total = 24.0*swr/hrs
202  else
203 ! calculate based on top of atmosphere value and cloud cover
204 ! - also calculated from Forest ETP
205  coscst = 1.0-1.3614*cos(rlat)
206  alpha = 18.0 - 64.884*coscst
207  alpha = alpha*4.1842*1.0d4 ! convert from cal/cm to J/m2
208  alpha = alpha/(24.0*3600.0) ! convert into w/m2
209  beta = 0.682 - 0.3183*coscst
210  sigma = 0.02*log(max(cloud,0.001)) + 0.03259
211 
212  total = toa*(beta-sigma*cloud*10.0) - alpha
213  endif
214 
215  if (subd_par) then
216 ! sub-daily variation in SWR
217 
218 ! calculate using hour angle (h)
219 ! - scale by (instantaneous insolation at h)/(mean daylight insolation) law of cosines
220 ! - when t = 0, h is 0 and is solar noon
221 ! - when t = total_t, h is the dawn angle i.e. sunrise
222  h = dawn_angle * real(t)/real(total_t)
223 ! calculate cos of the zenith angle
224  cos_zen = sin(rlat)*sin(del) + cos(rlat)*cos(del)*cos(h)
225 ! calculate instantaneous insolation at hour angle h
226  toa_h = solar_const * cos_zen
227 ! toa_h = solar_const * ( sin(rlat)*sin(del) +
228 ! & cos(rlat)*cos(del)*cos(h) )
229 ! if(day==160) print*, h,toa,toa_h,toa_h/toa
230 ! scale SWR (total) from daytime mean to instantaneous
231 ! - scale by the ratio of toa instantaneous insolation to daytime mean
232  if(toa>0.0) then
233  total = total*toa_h/toa
234  else
235  total = 0.0
236  endif
237 ! if(day==160) print*, total
238 ! reset top of atmosphere to instantaneous value
239  toa = toa_h
240 
241  endif
242 
243  if (total<0.0) total=0.0
244  if(toa<=0.0) then
245  clearness = 0.0
246  else
247  clearness = total/toa
248  endif
249 
250 ! calculate diffuse irradiance (from Spitters etal 1986)
251  if(subd_par) THEN
252 ! hourly data
253  dj_r = 0.847 - 1.61*cos_zen + 1.04*cos_zen**2
254  dj_k = (1.47-dj_r)/1.66
255  if(clearness<0.22) THEN
256  diffprop = 1.0
257  ELSE IF(clearness<0.35) THEN
258  diffprop = 1.0-6.4*(clearness-0.22)**2
259  ELSE IF(clearness<dj_k) THEN
260  diffprop = 1.47-1.66*clearness
261  ELSE
262  diffprop = dj_r
263  endIF
264  ELSE
265 ! daily data
266  if(clearness<0.07) THEN
267  diffprop = 1.0
268  ELSE IF(clearness<0.35) THEN
269  diffprop = 1.0-2.3*(clearness-0.07)**2
270  ELSE IF(clearness<0.75) THEN
271  diffprop = 1.33-1.46*clearness
272  ELSE
273  diffprop = 0.23
274  endIF
275  endIF
276 
277  diffuse = total*diffprop
278 
279 ! PAR is about 48% of the total irradiance. A better formula could be found.
280  total = 0.48*total
281  diffuse = 0.48*diffuse ! not sure this is really right
282  ! ... (Ghislain)
283 ! convert to mol/m2/sec from W/m2
284  total = 4.6d-6*total
285  diffuse = 4.6d-6*diffuse
286  direct = total - diffuse
287 
288 else
289 
290  total = 0.0
291  diffuse = 0.0
292  direct = 0.0
293 
294 endif
295 
296 ! set zenith angle to zero if requested (originaly SDGVM default)
297 if (.not.calc_zen) cos_zen = 1.0
298 
299 end subroutine pfd_ant
