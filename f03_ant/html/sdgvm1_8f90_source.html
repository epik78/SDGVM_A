79  enddo

192  ssp%mnthprc(mnth)=ssp%mnthprc(mnth)+prc(mnth,day)/no_days(year+nn1+nn2,mnth,thty_dys)

198  enddo
199 enddo
200

202  do day=1,no_days(year+nn1+nn2,mnth,thty_dys)
203  if (hum(mnth,day)<30.0) hum(mnth,day) = 30.0
204  if (hum(mnth,day)>95.0) hum(mnth,day) = 95.0
205  ssp%mnthhum(mnth)=ssp%mnthhum(mnth)+hum(mnth,day)/no_days(year+nn1+nn2,mnth,thty_dys)
206  enddo
207 enddo
208

210  ssp%emnthtmp(:,nn1+1)=ssp%mnthtmp(:)
211  ssp%emnthprc(:,nn1+1)=ssp%mnthprc(:)
212  ssp%emnthhum(:,nn1+1)=ssp%mnthhum(:)
213 ELSE
214  ssp%emnthtmp(:,nn1+1)=0.95*ssp%emnthtmp(:,nn1+1)+0.05*ssp%mnthtmp(:)
215  ssp%emnthprc(:,nn1+1)=0.95*ssp%emnthprc(:,nn1+1)+0.05*ssp%mnthprc(:)
216  ssp%emnthhum(:,nn1+1)=0.95*ssp%emnthhum(:,nn1+1)+0.05*ssp%mnthhum(:)
217 endIF
